def save(data, id=nil)
  id ||= @next_id += 1

  @records[id] = data

  MemoryRecord.new(data.to_hash.merge(id: id))
end

def where(params)
  matches = @records.select do |id, record|
    same_id = (params.delete(:id) || id) == id

    params.all? { |attr, val| record.send(attr) == val } && same_id
  end

  Relation.new(matches)
end
